# Lazy load variables

Trainer contains four variable saver, rnd, logger and writter(A SummaryWritter object in Tensorboard) so that you can directly use them. All these variables are lazy-load instances.

from thexp import Trainer,Params
import random

class myTrainer(Trainer):
    pass

trainer = myTrainer(Params())

for i in range(50):
    trainer.logger.info(i)

for i in range(20):
    trainer.writter.add_scalar("test",random.random(),i)

    def __init__(self):
        super().__init__()
        self.fc = nn.Linear(28 * 28, 10)

    def forward(self, x):
        x = x.view(x.shape[0], -1)
        x = self.fc(x)
        return x

# Trainer

Define a subclass of Trainer, and extends/override callbacks/datasets/models/train_batch methods.

from thexp.frame import Meter, Params, Trainer
class MyTrainer(Trainer):
    def callbacks(self, params: Params):
        from thexp import callbacks
        callbacks.LoggerCallback().hook(self) # Add callback for automaticly logging

    def datasets(self, params: Params):
        from torchvision import transforms
        from torchvision.datasets import FakeData
        from thexp.torch.data.dataloader import DataLoader

        dataset = FakeData(size=32 * 10, image_size=(28, 28), transform=transforms.ToTensor())
        train_loader = eval_loader = test_loader = DataLoader(dataset, shuffle=True, batch_size=32, drop_last=True)

        self.regist_databundler(
            train=train_loader,
            test=test_loader,
            eval=eval_loader,
        )
        self.to(self.device)

    def models(self, params: Params):
        from torch.optim import SGD
        self.model = MyModel()
        self.optim = SGD(self.model.parameters(), lr=params.lr)
        self.cross = nn.CrossEntropyLoss()
        self.to(self.device)

    def train_batch(self, eidx, idx, global_step, batch_data, params, device):
        optim, cross = self.optim, self.cross
        meter = Meter()
        xs, ys = batch_data

        # Training logic
        logits = self.model(xs)
        meter.loss = cross(logits, ys)

        # Back propagation
        meter.loss.backward()
        optim.step()
        optim.zero_grad()

        return meter


params = Params()
params.epoch=20
params.lr = 0.01

trainer = MyTrainer(params)
trainer.train()

# Auto device

You can see that the code above call to(self.device) in the final of the methods datasets() and models(). Like model.to(device) or tensor.to(device), when you call this method in trainer, the data and model will be automatically allocated the assigned device.

The device(A torch.device object) in Trainer is generated from the device(A built-in str object) in Params.

# Callbacks
