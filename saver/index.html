# Store model or checkpoint : Saver

from thexp import Saver
1

This module is used for store model. Three saving methods were defined for diffierent actions.

All store/restore functions is a wrap of torch.save/torch.load.

  • checkpoint : Most used format. Saver will keeps latest max_to_keep chackpoints, others will be deleted. File will have a format as {step}.ckpt.
  • keypoint : All files will be keeped. This form will save a checkpoint that not be affected by max_to_keep ，File will have a format as keep.{step}.ckpt。
  • model : Like keypoint, but have a name format as model.{step}.pth.
from thexp import Saver

saver = Saver("./sav",max_to_keep=3)
for i in range(10):
    saver.save_keypoint(i,{"a":i},{"b":i})
for i in range(10):
    saver.save_checkpoint(i,{"a":i},{"b":i})
for i in range(10):
    saver.save_model(i,{"a":i},{"b":i})

print(saver.find_keypoints())
print(saver.find_checkpoints())
print(saver.find_models())
